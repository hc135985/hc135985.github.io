这个功能的主要的难点在拖拽这一块，没做过类似功能的肯定还是有点懵逼的。

要监听我们上传范围内的文件拖入、拖出、松开左键、滑动事件。

不bb，上代码

/**
 * 文件拖拽上传
 * @param {Number} count 单次上传个数  defaultValue => 1
 * @param {Function} btnEnter 划入回调 
 * @param {Function} btnLeave 划出/松开鼠标回调
 * @param {Function} onUpload 上传回调 参数 => FileList
 * @returns
 */
const FilesDragAndDrop = ({
    count = 1,
    btnEnter,
    btnLeave,
    onUpload,
    children
}) => {
  const drop = useRef();

  const handleDragOver = e => {
    e.preventDefault();
    e.stopPropagation();
    btnLeave && btnEnter();
  };

  const handleDrop = e => {
    e.preventDefault();
    e.stopPropagation();
    const files = [...e.dataTransfer.files];
    btnLeave && btnLeave();
    if (count && count < files.length) {
      Message.warning(`抱歉，每次最多只能上传${count} 文件。`, 'error', 2000);
      return;
    }
    onUpload(files);
    btnLeave && btnLeave();
  };

  const handleDragEnter = e => {
    e.preventDefault();
    e.stopPropagation();
    btnEnter && btnEnter();
  };

  const handleDragLeave = e => {
    e.preventDefault();
    e.stopPropagation();
    btnLeave && btnLeave();
  };

  useEffect(() => {
    drop.current.addEventListener('dragover', handleDragOver);
    drop.current.addEventListener('drop', handleDrop);
    drop.current.addEventListener('dragenter', handleDragEnter);
    drop.current.addEventListener('dragleave', handleDragLeave);
    return () => {
      drop.current.removeEventListener('dragover', handleDragOver);
      drop.current.removeEventListener('drop', handleDrop);
      drop.current.removeEventListener('dragenter', handleDragEnter);
      drop.current.removeEventListener('dragleave', handleDragLeave);
    };
  });

  return (
    <div ref={drop} className={classList.FilesDragAndDrop}>
      {children}
    </div>
  );
};

// 简单的调用
<FilesDragAndDrop
  onUpload={onUpload}
  btnLeave={btnLeave}
  btnEnter={btnEnter}
>
  <div className={styles.uploadTitle}>将你的简历拖拽到此处自动解析</div>
  <div
    className={styles.uploadBtn}
    onMouseLeave={btnLeave}
    onMouseEnter={btnEnter}
  >
    选择文件
    <input
      onChange={onChange}
      title={`支持${ALLOW_EXT_TEXT}格式文件，大小不超过${maxSize}M`}
      size="3"
      name="newResume"
      type="file"
    />
  </div>
  <div className={styles.uploadText}>
    支持word、pdf、ppt、txt、wps格式文件，文件大小控制在10M以内
  </div>
</FilesDragAndDrop>
